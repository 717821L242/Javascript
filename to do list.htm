    // body { font-family: Arial, sans-serif; text-align: center; } .todo-container { max-width: 400px; margin: 0 auto; } ul { list-style-type: none; padding: 0; } li { display: flex; align-items: center; justify-content: space-between; margin-bottom: 5px; } .done { text-decoration: line-through; color: #888; } function addTask() { var taskInput = document.getElementById("taskInput"); var taskText = taskInput.value.trim(); if (taskText === "") { return; } var taskList = document.getElementById("taskList"); var taskItem = document.createElement("li"); taskItem.innerHTML = ` ${taskText} `; taskList.appendChild(taskItem); taskInput.value = ""; } function markAsDone(button) { var taskSpan = button.previousElementSibling; taskSpan.classList.toggle("done"); } function removeTask(button) { var taskItem = button.parentNode; var taskList = document.getElementById("taskList"); taskList.removeChild(taskItem); }
